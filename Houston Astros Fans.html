# Fixing the date
for count in range(len(ATTENDENCE['GAME_DT'])):
    bad_date = str(ATTENDENCE['GAME_DT'][count])
    year = bad_date[0:4]
    month = bad_date[4:6]
    day = bad_date[6:8]
    new_date = year + "-" + month + "-" + day
    ATTENDENCE.at[count, 'GAME_DT'] = pd.to_datetime(new_date)
    ATTENDENCE.at[count, 'YEAR'] = year
